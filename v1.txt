import React, { useState, useMemo, useEffect } from 'react';
import { 
  LayoutDashboard, 
  Package, 
  TrendingUp, 
  Cpu, 
  Settings, 
  Search, 
  Bell, 
  AlertTriangle, 
  ArrowUpRight, 
  MessageSquare, 
  Send, 
  X, 
  Check, 
  RefreshCw, 
  Sparkles, 
  Filter, 
  Plus, 
  Trash2, 
  Database,
  Clock
} from 'lucide-react';
import { 
  BarChart, 
  Bar, 
  XAxis, 
  YAxis, 
  Tooltip, 
  ResponsiveContainer, 
  Cell 
} from 'recharts';

export default function App() {
  // --- STATE MANAGEMENT ---
  const [activeTab, setActiveTab] = useState('Dashboard');
  const [searchQuery, setSearchQuery] = useState('');
  const [categoryFilter, setCategoryFilter] = useState('All');
  const [isChatOpen, setIsChatOpen] = useState(false);
  const [chatInput, setChatInput] = useState('');
  const [notifications, setNotifications] = useState([
    { id: 1, text: "AI detected abnormal demand spike for 'SmartHub Pro'", time: "5m ago", unread: true },
    { id: 2, text: "Apex Charger stock dipped below critical safety threshold", time: "2h ago", unread: true }
  ]);
  const [showNotifications, setShowNotifications] = useState(false);
  const [toasts, setToasts] = useState([]);

  // --- INITIAL INVENTORY DATA (State-Driven) ---
  const [inventory, setInventory] = useState([
    { id: 1, name: 'SmartHub Pro', sku: 'SH-901', category: 'Electronics', stock: 12, maxStock: 150, price: 129.99, status: 'Low Stock' },
    { id: 2, name: 'AeroGlow Bulb', sku: 'AG-204', category: 'Smart Home', stock: 145, maxStock: 200, price: 19.99, status: 'Optimal' },
    { id: 3, name: 'Nebula Router', sku: 'NR-505', category: 'Networking', stock: 8, maxStock: 80, price: 189.50, status: 'Low Stock' },
    { id: 4, name: 'Quantum Pad', sku: 'QP-112', category: 'Accessories', stock: 84, maxStock: 120, price: 49.99, status: 'Optimal' },
    { id: 5, name: 'SyncBand v2', sku: 'SB-883', category: 'Wearables', stock: 230, maxStock: 300, price: 79.99, status: 'Optimal' },
    { id: 6, name: 'Apex Charger', sku: 'AC-302', category: 'Accessories', stock: 5, maxStock: 100, price: 29.99, status: 'Low Stock' },
    { id: 7, name: 'Ember Smart Mug', sku: 'EM-401', category: 'Smart Home', stock: 67, maxStock: 150, price: 99.95, status: 'Optimal' },
    { id: 8, name: 'Vortex VR Headset', sku: 'VX-770', category: 'Electronics', stock: 11, maxStock: 50, price: 499.00, status: 'Low Stock' }
  ]);

  // --- REBUY RECOMMENDATIONS (State-Driven) ---
  const [recommends, setRecommends] = useState([
    { id: 'rec-1', name: 'SmartHub Pro', currentStock: 12, predictedSales: 45, recommendQty: 40, status: 'Pending' },
    { id: 'rec-2', name: 'Nebula Router', currentStock: 8, predictedSales: 30, recommendQty: 25, status: 'Pending' },
    { id: 'rec-3', name: 'Apex Charger', currentStock: 5, predictedSales: 60, recommendQty: 55, status: 'Pending' }
  ]);

  // --- CHAT DIALOGUE STATE ---
  const [chatMessages, setChatMessages] = useState([
    { sender: 'ai', text: "Hello! I am SmartStock AI. I monitor your warehouse levels and demand velocities. How can I help you optimize your operations today?", time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) }
  ]);
  const [isTyping, setIsTyping] = useState(false);

  // --- NEW PRODUCT ADDITION FORM STATE ---
  const [showAddModal, setShowAddModal] = useState(false);
  const [newProduct, setNewProduct] = useState({
    name: '', sku: '', category: 'Electronics', stock: '', maxStock: '', price: ''
  });

  // --- HELPER FUNCTION FOR TOASTS ---
  const showToast = (message, type = 'success') => {
    const id = Date.now();
    setToasts(prev => [...prev, { id, message, type }]);
    setTimeout(() => {
      setToasts(prev => prev.filter(t => t.id !== id));
    }, 4000);
  };

  // --- DERIVED METRICS (DYNAMIC UPDATES) ---
  const metrics = useMemo(() => {
    const totalProducts = inventory.length;
    const lowStockAlerts = inventory.filter(item => item.stock < 20).length;
    const totalValue = inventory.reduce((acc, curr) => acc + (curr.stock * curr.price), 0);
    const insightsGenerated = recommends.filter(r => r.status === 'Pending').length + 138;
    
    return {
      totalProducts,
      lowStockAlerts,
      totalValue: totalValue.toLocaleString('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 }),
      insightsGenerated
    };
  }, [inventory, recommends]);

  // --- CATEGORY WISE SALES METRICS FOR CHART ---
  const salesByProductCategory = useMemo(() => {
    const defaultSalesMap = {
      'Electronics': 24800,
      'Smart Home': 18350,
      'Networking': 15120,
      'Accessories': 9400,
      'Wearables': 12900
    };
    
    return Object.keys(defaultSalesMap).map(cat => {
      const itemCount = inventory.filter(i => i.category === cat).length;
      return {
        category: cat,
        Sales: defaultSalesMap[cat] + (itemCount * 350)
      };
    });
  }, [inventory]);

  // --- FILTERED INVENTORY FOR THE DATA TABLE ---
  const filteredInventory = useMemo(() => {
    return inventory.filter(item => {
      const matchesSearch = item.name.toLowerCase().includes(searchQuery.toLowerCase()) || 
                            item.sku.toLowerCase().includes(searchQuery.toLowerCase());
      const matchesCategory = categoryFilter === 'All' || item.category === categoryFilter;
      return matchesSearch && matchesCategory;
    });
  }, [inventory, searchQuery, categoryFilter]);

  // --- ACTION HANDLERS ---
  const handleApproveReorder = (id, productName, qty) => {
    setRecommends(prev => prev.map(rec => rec.id === id ? { ...rec, status: 'Approved' } : rec));
    
    setInventory(prev => prev.map(item => {
      if (item.name === productName) {
        const newStock = item.stock + qty;
        return {
          ...item,
          stock: newStock,
          status: newStock >= 20 ? 'Optimal' : 'Low Stock'
        };
      }
      return item;
    }));

    showToast(`Replenishment order for ${qty}x ${productName} dispatched!`, 'success');
  };

  const handleManualRestock = (id, productName) => {
    setInventory(prev => prev.map(item => {
      if (item.id === id) {
        const added = 50;
        const newStock = item.stock + added;
        return {
          ...item,
          stock: newStock,
          status: newStock >= 20 ? 'Optimal' : 'Low Stock'
        };
      }
      return item;
    }));
    showToast(`Manually restocked 50 units of ${productName}`, 'info');
  };

  const handleAddNewProduct = (e) => {
    e.preventDefault();
    if (!newProduct.name || !newProduct.sku || !newProduct.stock || !newProduct.price) {
      showToast('Please fill out all product details', 'error');
      return;
    }

    const stockNum = parseInt(newProduct.stock);
    const maxStockNum = parseInt(newProduct.maxStock || 100);
    const priceNum = parseFloat(newProduct.price);

    const createdItem = {
      id: Date.now(),
      name: newProduct.name,
      sku: newProduct.sku,
      category: newProduct.category,
      stock: stockNum,
      maxStock: maxStockNum,
      price: priceNum,
      status: stockNum < 20 ? 'Low Stock' : 'Optimal'
    };

    setInventory(prev => [createdItem, ...prev]);
    setShowAddModal(false);
    setNewProduct({ name: '', sku: '', category: 'Electronics', stock: '', maxStock: '', price: '' });
    showToast(`Successfully added ${createdItem.name} to database`, 'success');
  };

  const handleDeleteProduct = (id, name) => {
    setInventory(prev => prev.filter(i => i.id !== id));
    showToast(`Removed ${name} from inventory catalog`, 'info');
  };

  // --- DYNAMIC AI CHAT ENGINE ---
  const handleSendMessage = (e) => {
    if (e) e.preventDefault();
    if (!chatInput.trim()) return;

    const userMsg = chatInput.trim();
    const timestamp = new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
    
    setChatMessages(prev => [...prev, { sender: 'user', text: userMsg, time: timestamp }]);
    setChatInput('');
    setIsTyping(true);

    setTimeout(() => {
      let aiResponse = "I have analyzed your inventory levels. ";
      const lowStockItems = inventory.filter(i => i.stock < 20);

      if (userMsg.toLowerCase().includes('stock') || userMsg.toLowerCase().includes('critical') || userMsg.toLowerCase().includes('low')) {
        if (lowStockItems.length > 0) {
          aiResponse += `You have ${lowStockItems.length} items with low stock warning alerts, including: ${lowStockItems.slice(0,3).map(i => i.name).join(', ')}. I highly recommend triggering the recommended Smart Reorders for replenishment.`;
        } else {
          aiResponse += "Remarkable! All your tracked catalog lines currently fulfill optimal warehouse safety buffers.";
        }
      } else if (userMsg.toLowerCase().includes('reorder') || userMsg.toLowerCase().includes('recommend') || userMsg.toLowerCase().includes('rebuy')) {
        const pendingRecs = recommends.filter(r => r.status === 'Pending');
        if (pendingRecs.length > 0) {
          aiResponse += `My predictive neural net recommends immediate orders for: ${pendingRecs.map(r => `${r.name} (+${r.recommendQty})`).join(', ')} based on a forecasted sales velocity of up to ${pendingRecs.reduce((acc,curr) => acc+curr.predictedSales, 0)} units this month.`;
        } else {
          aiResponse += "All active rebuy recommendations have already been initiated and queued for supplier fulfillment.";
        }
      } else if (userMsg.toLowerCase().includes('sales') || userMsg.toLowerCase().includes('top') || userMsg.toLowerCase().includes('category')) {
        aiResponse += "Electronics is currently leading your sales velocity with $24.8K processed, followed closely by Smart Home at $18.3K. These are driving roughly 63% of overall weekly capital turnover.";
      } else {
        aiResponse += "I can provide optimized replenishment recommendations, critical stock safety reports, or cross-reference seasonal supplier logs. Try asking me 'What is low in stock?' or 'Explain the rebuy suggestions'.";
      }

      setChatMessages(prev => [...prev, { sender: 'ai', text: aiResponse, time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) }]);
      setIsTyping(false);
    }, 1100);
  };

  const handleSuggestionClick = (suggestionText) => {
    setChatInput(suggestionText);
  };

  return (
    <div className="min-h-screen bg-[#0A0A0A] text-zinc-100 flex font-sans antialiased w-full">
      
      {/* ================= SIDEBAR ================= */}
      <aside className="w-64 bg-[#0F0F13] border-r border-[#27272A] flex flex-col fixed h-full z-30 transition-transform">
        <div className="p-6 border-b border-[#27272A] flex items-center gap-3">
          <div className="h-9 w-9 rounded-lg bg-gradient-to-br from-blue-500 to-purple-600 flex items-center justify-center shadow-[0_0_15px_rgba(59,130,246,0.4)]">
            <Cpu className="h-5 w-5 text-white animate-pulse" />
          </div>
          <div>
            <h1 className="text-md font-bold tracking-tight bg-gradient-to-r from-white via-zinc-200 to-zinc-400 bg-clip-text text-transparent">SmartStock AI</h1>
            <p className="text-[10px] text-blue-400 font-semibold uppercase tracking-wider">Enterprise v4.2</p>
          </div>
        </div>

        <nav className="flex-1 px-4 py-6 space-y-1.5">
          <p className="px-3 text-[10px] font-semibold text-zinc-500 uppercase tracking-wider mb-2">Core Hub</p>
          
          <button 
            onClick={() => setActiveTab('Dashboard')}
            className={`w-full flex items-center justify-between px-3 py-2.5 rounded-lg text-sm font-medium transition-all ${
              activeTab === 'Dashboard' 
                ? 'bg-gradient-to-r from-blue-500/10 to-transparent text-blue-400 border-l-2 border-blue-500' 
                : 'text-zinc-400 hover:text-white hover:bg-zinc-800/40'
            }`}
          >
            <div className="flex items-center gap-2.5">
              <LayoutDashboard className="h-4 w-4" />
              <span>Dashboard</span>
            </div>
            {activeTab === 'Dashboard' && <div className="h-1.5 w-1.5 rounded-full bg-blue-500 shadow-[0_0_8px_#3b82f6]" />}
          </button>

          <button 
            onClick={() => setActiveTab('Inventory')}
            className={`w-full flex items-center justify-between px-3 py-2.5 rounded-lg text-sm font-medium transition-all ${
              activeTab === 'Inventory' 
                ? 'bg-gradient-to-r from-blue-500/10 to-transparent text-blue-400 border-l-2 border-blue-500' 
                : 'text-zinc-400 hover:text-white hover:bg-zinc-800/40'
            }`}
          >
            <div className="flex items-center gap-2.5">
              <Package className="h-4 w-4" />
              <span>Inventory Catalog</span>
            </div>
            {metrics.lowStockAlerts > 0 && (
              <span className="text-[10px] bg-red-500/20 text-red-400 border border-red-500/30 px-1.5 py-0.5 rounded-full font-bold">
                {metrics.lowStockAlerts}
              </span>
            )}
          </button>

          <button 
            onClick={() => setActiveTab('Sales')}
            className={`w-full flex items-center justify-between px-3 py-2.5 rounded-lg text-sm font-medium transition-all ${
              activeTab === 'Sales' 
                ? 'bg-gradient-to-r from-blue-500/10 to-transparent text-blue-400 border-l-2 border-blue-500' 
                : 'text-zinc-400 hover:text-white hover:bg-zinc-800/40'
            }`}
          >
            <div className="flex items-center gap-2.5">
              <TrendingUp className="h-4 w-4" />
              <span>Sales & Velocity</span>
            </div>
          </button>

          <div className="pt-6">
            <p className="px-3 text-[10px] font-semibold text-zinc-500 uppercase tracking-wider mb-2">Intelligence</p>
            
            <button 
              onClick={() => setActiveTab('AI Insights')}
              className={`w-full flex items-center justify-between px-3 py-2.5 rounded-lg text-sm font-medium transition-all relative overflow-hidden group ${
                activeTab === 'AI Insights' 
                  ? 'bg-gradient-to-r from-purple-500/15 to-transparent text-purple-400 border-l-2 border-purple-500' 
                  : 'text-zinc-400 hover:text-white hover:bg-zinc-800/40'
              }`}
            >
              <div className="flex items-center gap-2.5 z-10">
                <Sparkles className="h-4 w-4 text-purple-400 group-hover:rotate-12 transition-transform" />
                <span>AI Insights</span>
              </div>
              <span className="text-[9px] bg-gradient-to-r from-blue-500 to-purple-600 text-white px-2 py-0.5 rounded font-extrabold shadow-[0_0_10px_rgba(139,92,246,0.3)]">
                PRO
              </span>
            </button>
          </div>

          <div className="pt-6">
            <p className="px-3 text-[10px] font-semibold text-zinc-500 uppercase tracking-wider mb-2">Systems</p>
            <button 
              onClick={() => setActiveTab('Settings')}
              className={`w-full flex items-center gap-2.5 px-3 py-2.5 rounded-lg text-sm font-medium transition-all ${
                activeTab === 'Settings' 
                  ? 'bg-gradient-to-r from-blue-500/10 to-transparent text-blue-400 border-l-2 border-blue-500' 
                  : 'text-zinc-400 hover:text-white hover:bg-zinc-800/40'
              }`}
            >
              <Settings className="h-4 w-4" />
              <span>Settings</span>
            </button>
          </div>
        </nav>

        <div className="p-4 border-t border-[#27272A] bg-zinc-950/40 flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="h-9 w-9 rounded-full bg-gradient-to-r from-blue-500 to-cyan-500 flex items-center justify-center text-zinc-950 font-bold text-sm shadow-md">
              JD
            </div>
            <div>
              <p className="text-xs font-semibold text-zinc-200">Julien Deaux</p>
              <p className="text-[10px] text-zinc-500">Warehouse Admin</p>
            </div>
          </div>
        </div>
      </aside>

      {/* ================= MAIN CONTAINER ================= */}
      <main className="flex-1 ml-64 min-h-screen flex flex-col relative pb-12">
        
        {/* ================= TOP HEADER ================= */}
        <header className="h-16 border-b border-[#27272A] bg-[#0A0A0A]/80 backdrop-blur-md sticky top-0 z-20 flex items-center justify-between px-8">
          <div className="flex items-center gap-4">
            <div className="relative w-80">
              <Search className="absolute left-3 top-2.5 h-4 w-4 text-zinc-500" />
              <input 
                type="text" 
                placeholder="Search products by code or name..." 
                value={searchQuery}
                onChange={(e) => setSearchQuery(e.target.value)}
                className="w-full bg-[#121217] text-xs text-zinc-200 pl-9 pr-4 py-2 rounded-lg border border-[#27272A] focus:outline-none focus:border-blue-500 focus:ring-1 focus:ring-blue-500 transition-all"
              />
            </div>
            
            {searchQuery && (
              <button 
                onClick={() => setSearchQuery('')}
                className="text-xs text-zinc-400 hover:text-white bg-zinc-800 px-2.5 py-1 rounded-full transition-all flex items-center gap-1"
              >
                Clear Search <X className="h-3 w-3" />
              </button>
            )}
          </div>

          <div className="flex items-center gap-4">
            <div className="hidden md:flex items-center gap-2 px-3 py-1 bg-zinc-900 border border-zinc-800 rounded-full text-xs text-zinc-400">
              <span className="h-2 w-2 rounded-full bg-emerald-500 animate-pulse" />
              <span>SmartSync Active</span>
            </div>

            <div className="relative">
              <button 
                onClick={() => setShowNotifications(!showNotifications)}
                className="p-2 rounded-lg bg-zinc-900 border border-zinc-800 hover:bg-zinc-800 text-zinc-300 relative transition-all"
              >
                <Bell className="h-4.5 w-4.5" />
                {notifications.length > 0 && (
                  <span className="absolute -top-1 -right-1 h-2 w-2 rounded-full bg-blue-500 shadow-[0_0_8px_#3b82f6]" />
                )}
              </button>

              {showNotifications && (
                <div className="absolute right-0 mt-2 w-80 bg-[#0F0F13] border border-[#27272A] rounded-xl shadow-xl z-50 overflow-hidden">
                  <div className="p-4 border-b border-zinc-800 flex justify-between items-center bg-zinc-950/50">
                    <h3 className="text-xs font-bold text-zinc-200">System Notifications</h3>
                    <button 
                      onClick={() => setNotifications([])} 
                      className="text-[10px] text-zinc-400 hover:text-blue-400 transition-colors"
                    >
                      Clear All
                    </button>
                  </div>
                  <div className="divide-y divide-zinc-800/60 max-h-64 overflow-y-auto">
                    {notifications.length === 0 ? (
                      <div className="p-6 text-center text-xs text-zinc-500">
                        No critical alerts to check
                      </div>
                    ) : (
                      notifications.map(notif => (
                        <div key={notif.id} className="p-3.5 hover:bg-zinc-900 transition-colors">
                          <p className="text-xs text-zinc-300 font-medium">{notif.text}</p>
                          <div className="flex items-center gap-2 mt-1.5">
                            <span className="text-[9px] text-zinc-500">{notif.time}</span>
                            <span className="h-1 w-1 rounded-full bg-blue-400" />
                            <span className="text-[9px] text-blue-400 font-semibold">Priority</span>
                          </div>
                        </div>
                      ))
                    )}
                  </div>
                </div>
              )}
            </div>

            <button 
              onClick={() => {
                setIsChatOpen(true);
                handleSuggestionClick("Generate fresh reorder list.");
                showToast("AI Audit Command initialized!", "info");
              }}
              className="flex items-center gap-1.5 px-3 py-1.5 rounded-lg bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-500 hover:to-purple-500 text-xs font-bold text-white shadow-[0_0_15px_rgba(139,92,246,0.3)] transition-all transform active:scale-95"
            >
              <Sparkles className="h-3.5 w-3.5" />
              <span>Ask SmartStock</span>
            </button>
          </div>
        </header>

        {/* ================= CONTENT CONTENT ================= */}
        <div className="flex-1 px-8 py-6">

          {activeTab === 'Dashboard' && (
            <>
              <div className="flex flex-col md:flex-row md:items-center justify-between mb-8 gap-4">
                <div>
                  <h2 className="text-2xl font-bold tracking-tight text-white flex items-center gap-2.5">
                    Operations Hub
                    <span className="px-2 py-0.5 text-[10px] bg-blue-500/10 text-blue-400 border border-blue-500/20 rounded font-semibold tracking-wide">
                      Live Warehouse
                    </span>
                  </h2>
                  <p className="text-xs text-zinc-400 mt-1">Real-time demand metrics, predictive restocking triggers, and cognitive warehouse optimization.</p>
                </div>
                
                <div className="flex items-center gap-3">
                  <button 
                    onClick={() => {
                      showToast("Synchronizing with centralized logistics standard ERP...", "info");
                    }}
                    className="flex items-center gap-2 px-3 py-2 bg-zinc-900 border border-zinc-800 hover:bg-zinc-800 rounded-lg text-xs font-medium text-zinc-300 transition-colors"
                  >
                    <RefreshCw className="h-3.5 w-3.5 text-zinc-400" />
                    <span>Resync Systems</span>
                  </button>

                  <button 
                    onClick={() => setShowAddModal(true)}
                    className="flex items-center gap-2 px-3.5 py-2 bg-blue-600 hover:bg-blue-500 rounded-lg text-xs font-bold text-white transition-all shadow-[0_4px_12px_rgba(59,130,246,0.2)]"
                  >
                    <Plus className="h-3.5 w-3.5" />
                    <span>Add New Line</span>
                  </button>
                </div>
              </div>

              {/* ================= TOP ROW: KPI CARDS ================= */}
              <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-6 mb-8">
                
                <div className="bg-[#0F0F13] border border-[#27272A] rounded-xl p-5 relative overflow-hidden group hover:border-zinc-700 transition-all">
                  <div className="flex items-center justify-between mb-4">
                    <span className="text-xs font-semibold text-zinc-400">Total Catalog Lines</span>
                    <div className="p-2 rounded-lg bg-blue-500/10 text-blue-400 border border-blue-500/10">
                      <Package className="h-4 w-4" />
                    </div>
                  </div>
                  <div className="flex items-baseline gap-2">
                    <span className="text-2xl font-bold tracking-tight">{metrics.totalProducts}</span>
                    <span className="text-[10px] font-semibold text-emerald-400 flex items-center gap-0.5">
                      <ArrowUpRight className="h-3 w-3" /> +12% MoM
                    </span>
                  </div>
                  <p className="text-[10px] text-zinc-500 mt-1">Sourced across 5 distinct categories</p>
                </div>

                <div className="bg-[#0F0F13] border border-[#27272A] rounded-xl p-5 relative overflow-hidden group hover:border-zinc-700 transition-all">
                  <div className="flex items-center justify-between mb-4">
                    <span className="text-xs font-semibold text-zinc-400">Low Stock Alerts</span>
                    <div className={`p-2 rounded-lg ${metrics.lowStockAlerts > 0 ? 'bg-red-500/10 text-red-400 border border-red-500/15 animate-pulse' : 'bg-zinc-800 text-zinc-400'}`}>
                      <AlertTriangle className="h-4 w-4" />
                    </div>
                  </div>
                  <div className="flex items-baseline gap-2">
                    <span className={`text-2xl font-bold tracking-tight ${metrics.lowStockAlerts > 0 ? 'text-red-400' : 'text-zinc-200'}`}>{metrics.lowStockAlerts}</span>
                    <span className="text-[10px] font-semibold text-emerald-400 flex items-center gap-0.5">
                      -3 since yesterday
                    </span>
                  </div>
                  <p className="text-[10px] text-zinc-500 mt-1">Requires immediate manual override or approval</p>
                </div>

                <div className="bg-[#0F0F13] border border-[#27272A] rounded-xl p-5 relative overflow-hidden group hover:border-zinc-700 transition-all">
                  <div className="flex items-center justify-between mb-4">
                    <span className="text-xs font-semibold text-zinc-400">Total Stock Value</span>
                    <div className="p-2 rounded-lg bg-zinc-800 text-zinc-300">
                      <TrendingUp className="h-4 w-4" />

                  <h4 className="text-sm font-bold text-white">Supplier Shipping Delay Warnings</h4>

                    Logistics analytics identify customs backlogs at Port of Newark. Safe backup times shifted from 7 days to 14 days.

                  <div className="mt-4 pt-4 border-t border-zinc-800 flex justify-between items-center text-[10px]">

                      Update Thresholds
                    </button>
                  </div>
                </div>

                <div className="bg-[#0F0F13] border border-zinc-800 p-5 rounded-xl">
                  <div className="flex items-center gap-2 mb-3">
                    <TrendingUp className="h-4.5 w-4.5 text-emerald-400" />

                    Stock turn rate for "Quantum Pad" slowed down by 14% over past quarter. Recommend bundle packages.
                  </p>
                  <div className="mt-4 pt-4 border-t border-zinc-800 flex justify-between items-center text-[10px]">
                    <span className="text-zinc-500">Est. capital unlock: $3,200</span>
                    <button 
                      onClick={() => showToast("Bundle configurations saved to store draft exports!", "success")} 
                      className="text-emerald-400 font-bold hover:text-emerald-300"

                      Draft Promotion
                    </button>
                  </div>
                </div>

              </div>
            </div>
          )}

          {/* ================= SETTINGS TAB VIEW ================= */}
          {activeTab === 'Settings' && (
            <div className="space-y-6">
              <div>
                <h2 className="text-xl font-bold text-white mb-1">Global System Preferences</h2>

              <div className="bg-[#0F0F13] border border-zinc-800 p-6 rounded-xl space-y-6">
                <div>
                  <h4 className="text-sm font-bold text-white mb-3">Auto-Reorder Level</h4>

                    <span className="text-xs font-semibold text-zinc-200">20 Units (Critical Buffer Safety)</span>
                  </div>
                </div>

                <div className="pt-6 border-t border-zinc-800/60">
                  <h4 className="text-sm font-bold text-white mb-3">Integrated APIs</h4>
                  <p className="text-xs text-zinc-400 mb-4">Sync inventory directly across external platforms.</p>
                  <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                    <div className="p-3 bg-zinc-950 border border-zinc-800 rounded-lg flex items-center justify-between">
                      <span className="text-xs font-bold text-zinc-300">Shopify API Sync</span>
                      <span className="px-2 py-0.5 rounded text-[10px] bg-emerald-500/10 text-emerald-400 border border-emerald-500/20 font-bold">Connected</span>
                    </div>
                    <div className="p-3 bg-zinc-950 border border-zinc-800 rounded-lg flex items-center justify-between">
                      <span className="text-xs font-bold text-zinc-300">Amazon Fulfillment API</span>
                      <span className="px-2 py-0.5 rounded text-[10px] bg-emerald-500/10 text-emerald-400 border border-emerald-500/20 font-bold">Connected</span>
                    </div>
                  </div>
                </div>

                <button 
                  onClick={() => showToast("Configuration presets updated and applied worldwide!", "success")}
                  className="px-4 py-2 bg-blue-600 hover:bg-blue-500 text-xs font-bold text-white rounded-lg transition-colors"
                >
                  Save System Variables
                </button>
              </div>
            </div>
          )}

        </div>

        {/* ================= FOOTER META ================= */}
        <footer className="mt-auto pt-6 border-t border-[#27272A]/40 text-center text-[10px] text-zinc-600">

      </main>

      {/* ================= FLOATING AI CHATBOT ================= */}
      <div className="fixed bottom-6 right-6 z-50 flex flex-col items-end">
        {isChatOpen && (
          <div className="w-80 md:w-96 bg-[#0F0F13] border border-[#27272A] rounded-2xl shadow-2xl flex flex-col overflow-hidden mb-4 transform scale-100 transition-all">
            <div className="bg-gradient-to-r from-blue-600 to-purple-600 p-4 flex justify-between items-center text-white">
              <div className="flex items-center gap-2">
                <Sparkles className="h-4 w-4" />
                <div>
                  <h3 className="text-xs font-bold tracking-wide">SmartStock AI Copilot</h3>

                </div>
              </div>
              <button 
                onClick={() => setIsChatOpen(false)}
                className="p-1 rounded-full hover:bg-white/10 transition-colors text-white"

                <X className="h-4.5 w-4.5" />
              </button>

            <div className="flex-1 p-4 h-64 overflow-y-auto space-y-3.5 bg-zinc-950/70 select-text">
              {chatMessages.map((msg, index) => (
                <div key={index} className={`flex ${msg.sender === 'user' ? 'justify-end' : 'justify-start'}`}>
                  <div className={`max-w-[85%] rounded-xl px-3 py-2 text-xs leading-normal ${
                    msg.sender === 'user' 
                      ? 'bg-blue-600 text-white rounded-tr-none' 
                      : 'bg-zinc-900 border border-zinc-800 text-zinc-200 rounded-tl-none'
                  }`}>
                    {msg.sender !== 'user' && (
                      <span className="block text-[8px] font-bold text-purple-400 mb-0.5 uppercase tracking-wider">AI Assistant</span>
                    )}
                    <p className="whitespace-pre-wrap">{msg.text}</p>
                    <span className="block text-[8px] text-zinc-500 text-right mt-1 font-mono">{msg.time}</span>
                  </div>
                </div>
              ))}
              
              {isTyping && (
                <div className="flex justify-start">
                  <div className="bg-zinc-900 border border-zinc-800 rounded-xl rounded-tl-none px-3 py-2.5 text-xs text-zinc-400">
                    <div className="flex items-center gap-1">
                      <span className="h-1.5 w-1.5 rounded-full bg-purple-400 animate-bounce" style={{animationDelay:'0ms'}} />
                      <span className="h-1.5 w-1.5 rounded-full bg-purple-400 animate-bounce" style={{animationDelay:'150ms'}} />
                      <span className="h-1.5 w-1.5 rounded-full bg-purple-400 animate-bounce" style={{animationDelay:'300ms'}} />
                    </div>
                  </div>
                </div>
              )}
            </div>

            <div className="px-3 py-2 border-t border-[#27272A]/60 bg-[#0F0F13] flex flex-wrap gap-1.5">
              <button 
                onClick={() => handleSuggestionClick("Which products are currently critical?")}
                className="text-[9px] bg-zinc-900 border border-zinc-800 hover:border-blue-500 text-zinc-300 hover:text-white px-2 py-1 rounded transition-colors text-left"
              >
                Check Critical Stock
              </button>
              <button 
                onClick={() => handleSuggestionClick("Predict next month's best sellers.")}
                className="text-[9px] bg-zinc-900 border border-zinc-800 hover:border-purple-500 text-zinc-300 hover:text-white px-2 py-1 rounded transition-colors text-left"
              >
                Sales Forecast
              </button>
              <button 
                onClick={() => handleSuggestionClick("Generate fresh reorder list.")}
                className="text-[9px] bg-zinc-900 border border-zinc-800 hover:border-blue-500 text-zinc-300 hover:text-white px-2 py-1 rounded transition-colors text-left"
              >
                Reorder Audit
              </button>
            </div>

            <form onSubmit={handleSendMessage} className="p-3 border-t border-[#27272A] bg-zinc-950 flex items-center gap-2">
              <input 
                type="text" 
                placeholder="Ask me something about stock velocity..."
                value={chatInput}
                onChange={(e) => setChatInput(e.target.value)}
                className="flex-1 bg-[#121217] text-xs text-zinc-200 px-3 py-2 rounded-lg border border-zinc-800 focus:outline-none focus:border-purple-500 transition-colors"
              />
              <button 
                type="submit"
                className="p-2 rounded-lg bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-500 hover:to-purple-500 text-white shadow-md active:scale-95 transition-transform"
              >
                <Send className="h-3.5 w-3.5" />
              </button>
            </form>
          </div>
        )}

        <button 
          onClick={() => setIsChatOpen(!isChatOpen)}
          className={`h-14 w-14 rounded-full flex items-center justify-center text-white shadow-2xl relative overflow-hidden transition-transform active:scale-95 group ${
            isChatOpen 
              ? 'bg-zinc-800 hover:bg-zinc-700' 
              : 'bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-500 hover:to-purple-500 shadow-[0_0_20px_rgba(139,92,246,0.4)]'
          }`}
          title="Toggle SmartStock AI Chat"
        >
          {!isChatOpen && (
            <span className="absolute inset-0 h-full w-full rounded-full bg-blue-500/20 animate-ping z-0" />
          )}
          {isChatOpen ? (
            <X className="h-6 w-6 relative z-10" />
          ) : (
            <MessageSquare className="h-6 w-6 relative z-10" />
          )}
        </button>
      </div>

      {/* ================= MODAL: ADD PRODUCT OVERLAY ================= */}
      {showAddModal && (
        <div className="fixed inset-0 bg-[#000]/80 backdrop-blur-sm z-50 flex items-center justify-center p-4">
          <div className="bg-[#0F0F13] border border-[#27272A] rounded-2xl w-full max-w-md overflow-hidden shadow-2xl">
            
            <div className="bg-zinc-950 p-5 border-b border-[#27272A] flex justify-between items-center">
              <div>
                <h3 className="text-sm font-bold text-white flex items-center gap-1.5">
                  <Database className="h-4.5 w-4.5 text-blue-400" /> Register Catalog Entry
                </h3>
                <p className="text-[10px] text-zinc-400 mt-0.5">Input system metrics to instantly track inventory assets</p>
              </div>
              <button onClick={() => setShowAddModal(false)} className="text-zinc-500 hover:text-white p-1">
                <X className="h-4.5 w-4.5" />
              </button>
            </div>

            <form onSubmit={handleAddNewProduct} className="p-6 space-y-4">
              <div>
                <label className="block text-[10px] text-zinc-400 font-bold uppercase tracking-wide mb-1.5">Product Name</label>
                <input 
                  type="text" 
                  placeholder="e.g., Nova Smart Charger" 
                  value={newProduct.name}
                  onChange={(e) => setNewProduct(prev => ({ ...prev, name: e.target.value }))}
                  className="w-full bg-[#121217] border border-[#27272A] rounded-lg px-3 py-2 text-xs text-zinc-100 focus:outline-none focus:border-blue-500 transition-colors"
                />
              </div>

              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="block text-[10px] text-zinc-400 font-bold uppercase tracking-wide mb-1.5">SKU ID</label>
                  <input 
                    type="text" 
                    placeholder="e.g., NV-109" 
                    value={newProduct.sku}
                    onChange={(e) => setNewProduct(prev => ({ ...prev, sku: e.target.value }))}
                    className="w-full bg-[#121217] border border-[#27272A] rounded-lg px-3 py-2 text-xs text-zinc-100 focus:outline-none focus:border-blue-500 transition-colors"
                  />
                </div>
                <div>
                  <label className="block text-[10px] text-zinc-400 font-bold uppercase tracking-wide mb-1.5">Category</label>
                  <select 
                    value={newProduct.category}
                    onChange={(e) => setNewProduct(prev => ({ ...prev, category: e.target.value }))}
                    className="w-full bg-[#121217] border border-[#27272A] rounded-lg px-3 py-2 text-xs text-zinc-100 focus:outline-none focus:border-blue-500"
                  >
                    <option>Electronics</option>
                    <option>Smart Home</option>
                    <option>Networking</option>
                    <option>Accessories</option>
                    <option>Wearables</option>
                  </select>
                </div>
              </div>

              <div className="grid grid-cols-3 gap-3">
                <div>
                  <label className="block text-[10px] text-zinc-400 font-bold uppercase tracking-wide mb-1.5">Initial Stock</label>
                  <input 
                    type="number" 
                    placeholder="e.g., 25" 
                    value={newProduct.stock}
                    onChange={(e) => setNewProduct(prev => ({ ...prev, stock: e.target.value }))}
                    className="w-full bg-[#121217] border border-[#27272A] rounded-lg px-3 py-2 text-xs text-zinc-100 focus:outline-none focus:border-blue-500"
                  />
                </div>
                <div>
                  <label className="block text-[10px] text-zinc-400 font-bold uppercase tracking-wide mb-1.5">Max Stock Cap</label>
                  <input 
                    type="number" 
                    placeholder="e.g., 100" 
                    value={newProduct.maxStock}
                    onChange={(e) => setNewProduct(prev => ({ ...prev, maxStock: e.target.value }))}
                    className="w-full bg-[#121217] border border-[#27272A] rounded-lg px-3 py-2 text-xs text-zinc-100 focus:outline-none focus:border-blue-500"
                  />
                </div>
                <div>
                  <label className="block text-[10px] text-zinc-400 font-bold uppercase tracking-wide mb-1.5">Unit Price ($)</label>
                  <input 
                    type="number" 
                    step="0.01" 
                    placeholder="e.g., 49.99" 
                    value={newProduct.price}
                    onChange={(e) => setNewProduct(prev => ({ ...prev, price: e.target.value }))}
                    className="w-full bg-[#121217] border border-[#27272A] rounded-lg px-3 py-2 text-xs text-zinc-100 focus:outline-none focus:border-blue-500"
                  />
                </div>
              </div>

              <div className="pt-4 flex justify-end gap-3 border-t border-[#27272A]/60">
                <button 
                  type="button" 
                  onClick={() => setShowAddModal(false)}
                  className="px-4 py-2 bg-transparent text-xs font-semibold text-zinc-400 hover:text-white transition-colors"
                >
                  Cancel
                </button>
                <button 
                  type="submit"
                  className="px-4 py-2 bg-blue-600 hover:bg-blue-500 text-xs font-bold text-white rounded-lg shadow-lg"
                >
                  Commit Product Line
                </button>
              </div>
            </form>

          </div>
        </div>
      )}

      {/* ================= TOASTS NOTIFICATIONS RENDERER ================= */}
      <div className="fixed top-6 left-1/2 transform -translate-x-1/2 z-50 flex flex-col gap-2 pointer-events-none">
        {toasts.map(toast => (
          <div 
            key={toast.id} 
            className="bg-[#0F0F13] border border-zinc-800 text-zinc-200 text-xs px-4 py-3 rounded-xl shadow-2xl flex items-center gap-2.5 pointer-events-auto transform translate-y-0 opacity-100 animate-bounce"
          >
            <div className={`h-2.5 w-2.5 rounded-full ${
              toast.type === 'error' ? 'bg-red-500' : toast.type === 'info' ? 'bg-blue-400' : 'bg-emerald-500'
            }`} />
            <span className="font-medium">{toast.message}</span>
            <button 
              onClick={() => setToasts(prev => prev.filter(t => t.id !== toast.id))}
              className="ml-3 text-zinc-500 hover:text-white pointer-events-auto"
            >
              <X className="h-3 w-3" />
            </button>
          </div>
        ))}
      </div>

    </div>
  );
}
